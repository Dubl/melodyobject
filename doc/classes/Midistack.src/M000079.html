  str=Midistack.ticker.to_s
  
        #~ harvestechostack
    unless @internaldone
        @stack.each {|d| 
        d.inc+=1

                if d.turned_on==false 
                  d.turned_on=true
                  str<<" "+d.val.to_s
                  $midi.note_on(d.val,d.chn,100)
                  if d.chn==0
                    e=d.uf

                    e.turned_on=false

                  #~ puts "#{d.val} #{d.chn} #{@ticker}"
                elsif d.inc>d.leng
                  $midi.note_off(d.val,d.chn,100)
                  d.del=true
                end
        }
              @stack.delete_if {|d| d.del}
    
              if @done

                  sleep($g)
              @stack.each {|d| $midi.note_off(d.val,d.chn,100)}
