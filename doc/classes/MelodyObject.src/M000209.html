    def backout
      
      if parent.nil?
        
        return
        
      else
        
        if self==@parent.children[parent.children.length-1]
          @status=:done
          @parent.status=:done
          @parent.backout
        else
          @status=:done
          @parent.currentchild+=1
          @parent.go_down
        end
        
      end
    end
